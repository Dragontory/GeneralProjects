PROGRAMSampleProgramIS
    INSTRUCTION TurnAround IS
        turnright
        turnright
    ENDTurnAround

    INSTRUCTION FindObstacle IS
        WHILE next-is-empty DO
            move
        END WHILE
    ENDFindObstacle

BEGIN
    WHILE true DO
        FindObstacle
        IF next-is-enemy THEN
            infect
        ELSE
            IF next-is-wall THEN
                turnleft
            ELSE
                TurnAround
            END IF
        END IF
    END WHILE
END SampleProgram
